# Funktion fÃ¼r wackelnden Text
shake() {
  local s="$1"
  local out=""
  for ((i=0;i<${#s};i++)); do
    out+="${s:i:1}"
    [[ $((RANDOM % 2)) -eq 0 ]] && out+=" "
  done
  echo "$out"
}
